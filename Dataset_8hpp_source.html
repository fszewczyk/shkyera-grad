125 Data Dataset::readLine(std::string &line) {
126  bool setLabel = false;
127  Vector<double> label;
128  Vector<double> data;
129 
130  std::string word;
131  std::stringstream stream(line);
132 
133  while (std::getline(stream, word, ',')) {
134  if (!setLabel) {
135  if (m_labelType == CONTINUOUS)
136  label = {1, utils::convert<double>(word)};
137  else if (m_labelType == CATEGORICAL) {
138  if (categoryToIndex.find(word) == categoryToIndex.end()) {
139  categoryToIndex.insert({word, categoryToIndex.size()});
140  indexToCategory.insert({categoryToIndex.size() - 1, word});
141  }
142 
143  label = Vector<double>::zeros(categoryToIndex.size());
144  label(categoryToIndex[word]) = 1;
145  }
146 
147  setLabel = true;
148  } else {
149  data.append(utils::convert<double>(word));
150  }
151  }
152 
153  return Data(label, data);
154 }

164 void Dataset::normalize() {
165  if (m_data.empty())
166  return;
167 
168  std::vector<double> dataMinimums(m_data[0].getData().getSize(), std::numeric_limits<double>::max());
169  std::vector<double> dataMaximums(m_data[0].getData().getSize(), std::numeric_limits<double>::lowest());
170 
171  for (auto d : m_data) {
172  auto data = d.getData();
173  for (size_t i = 0; i < data.getSize(); ++i) {
174  dataMinimums[i] = std::min(dataMinimums[i], data[i]);
175  dataMaximums[i] = std::max(dataMaximums[i], data[i]);
176  }
177  }
178 
179  for (auto &d : m_data) {
180  for (size_t i = 0; i < d.getData().getSize(); ++i) {
181  d.normalize(i, dataMinimums[i], dataMaximums[i]);
182  }
183  }
184 }
